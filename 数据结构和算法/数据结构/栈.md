# 栈
- 栈的特点是先进后出，后进先出
- 栈是一种'操作受限'的线性表，只允许在一端（栈顶）插入和删除数据
- 栈的可以使用数组或者链表实现

### 栈的常见操作
- push 压栈
- pop 出栈
- peek 查看栈顶元素

### 使用数组实现栈
~~~java
public class MyStack{
    private Integer[] arr;
    private int p = -1;

    public MyStack(int len){
        if (len <= 0) {
            throw new IllegalArgumentException("MyStack初始化大小必须大于0");
        }
        this.arr = new Integer[len];
    }

    public boolean push(Integer v){
        if(p == arr.length-1) return false;
        arr[++p] = v;
        return true;
    }

    public Integer pop(){
        if(p == -1) return null;
        Integer res = arr[p];
        arr[p--] = null;
        return res;
    }

    public Integer peek(){
        if(p == -1) return null;
        return arr[p];
    }

    public static void main(String[] args){
        MyStack myStack = new MyStack(2);
        System.out.println(myStack.push(1));
        System.out.println(myStack.push(2));
        System.out.println(myStack.push(3));

        System.out.println(myStack.peek());
        System.out.println(myStack.pop());
        System.out.println(myStack.peek());
        System.out.println(myStack.pop());

        System.out.println(myStack.peek());
    }
}
~~~

### 使用链表实现栈
~~~java
public class MyStack<T>{

    public static class Node<T>{
        public T value;
        public Node nextNode;
        public Node(T value){
            this.value = value;
        }
    }

    public Node head;
    public MyStack(){
    }
    
    public void push(T value){
        if(head == null){
            head = new Node(value);
        }else{
            Node p = head;
            head = new Node(value);
            head.nextNode = p;
        }
    }
        
    public T pop(){
        if(head == null) return null;
        T res = (T)head.value;
        head = head.nextNode;
        return res;
    }

    public T peek(){
        if(head == null) return null;
        return (T)head.value;
    }

    public static void main(String[] arg){
        MyStack myStack = new MyStack();
        myStack.push(1);
        myStack.push(2);
        System.out.println(myStack.peek());
        System.out.println(myStack.pop());
        System.out.println(myStack.peek());
        System.out.println(myStack.pop());
        System.out.println(myStack.peek());
        System.out.println(myStack.pop());
    }

}
~~~

~~~shell
# 常见的端口扫描工具 nmap ，安装命令 apt install nmap 或 yum install namp -y
# 注意事项
#     在进行网络扫描时，确保你有适当的授权。未经授权的端口扫描可能会被视为恶意行为。
#     扫描可能会被目标系统的防火墙或入侵检测系统检测到，因此在敏感环境中操作时要格外小心。
#     Nmap功能强大，但也有一定的复杂性。查阅Nmap官方文档了解更多选项和用法。
# nmap 使用
nmap [服务器IP或域名] [-p 80,443 或 1-40000] [-sV]
# -p 参数指定扫描端口或者是端口扫描范围
# -sV 参数显示端口服务的版本号
~~~

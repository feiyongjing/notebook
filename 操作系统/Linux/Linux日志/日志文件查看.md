# 如何查看日志
~~~shell
#查看文件内容
cat [文件名]
# -n参数，同时显示行号

#倒立显示文件内容
tac [文件名]

#部分文件内容查看，查看文件的开头部分
head -n 20 [文件名]
# -n参数指定显示文件的开头多少行，不加-n参数则默认显示前10行

#部分文件内容查看，查看文件的末尾部分
tail -fn 20 [文件名]
# -n参数指定显示文件的末尾多少行，不加-n参数则默认显示末尾10行
# -f参数动态显示文件的末尾

#长文件内容查看方式1
more [文件名]
# 进入查看后，使用空格或f向下翻页、回车向下换行、q或Q退出查看
# 缺点不能向上翻页和内容搜索

#长文件内容查看方式2
less [文件名]
# 进入查看后，使用空格或f向下翻页、pgUp和pgDn上下翻页、回车向下换行和上下箭头上下换行、q或Q退出查看
# 输入"/关键词"会搜索所有标记关键词，按n查找下一个关键词所在位置，按N查找上一个关键词所在位置
~~~

# 常见的系统日志

# /var/log/cron*            定时任务执行的日志文件
# /var/log/audit/audit.log  audit审计的日志文件
~~~

~~~shell
# 文件压缩为.gz压缩包，压缩后的文件后缀是 .gz ，压缩后原文件是不保留的
gzip [文件]
# -d参数表示解压缩文件

# .gz压缩包解压缩，解压缩后压缩包是不保留的
gunzip [压缩文件]

# 目录压缩和接压，压缩后的文件后缀是 .tar.gz，默认压缩后原目录会保留而解压后压缩包也会保留
tar -cvzf [打包压缩后的文件] [原目录]
# -c参数将目录打包，虽然打包后的文件可以以任意名称命名但是为了区分一般以.tar命名
# -C参数用于指定解压缩目录的路径
# -v参数显示目录下的详细信息，即显示目录下子目录和子文件
# -f参数指定打包或打包压缩后的文件名，注意-f参数必须是最后一个参数然后后面是压缩包名称，否则会报错
# -z参数打包的同时压缩为.gz，即将打好的.tar包压缩为.tar.gz，也可以不加-z参数将.tar包手动使用gzip压缩为.tar.gz文件
# -j参数打包的同时压缩为.bz2，即将打好的.tar包压缩为.tar.bz2，也可以不加-z参数将.tar包手动使用bzip2压缩为.tar.bz2文件
# -x参数解包，将-cvzf的-c参数替换为-x参数就是解压缩，例如 tar -xvzf test.tar.gz 或者 tar -xvf test.tar 或者 tar -xvjf test.tar.bz2

# linux和windows通用的压缩包zip的压缩
zip [压缩文件名] [原文件或目录]
# -r参数表示压缩目录

# linux和windows通用的压缩包zip的解压缩
unzip [压缩文件名]

# 文件压缩为.bz2格式，.bz2文件的压缩要比之前几种压缩压缩后的文件小，默认压缩后原文件是不保留的
bzip2 [文件]
# -k参数压缩后原文件保留

# .bz2压缩包解压缩，默认解压缩后压缩包是不保留的
bunzip2 [压缩文件]
# -k参数压缩后原文件保留
~~~

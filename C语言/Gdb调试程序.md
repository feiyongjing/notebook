~~~shell
# Gdb调试需要添加-g参数，以下是两种方式生成可执行程序
gcc -g hello.c -o hello
gcc -g -c hello.c && gcc -o hello hello.o

# gdb调试，执行命令进入调试命令窗口
gdb [需要调试的程序路径]

# 调试命令窗口添加运行程序运行参数
set args [参数1] [参数2] [更多参数...]

# 调试命令窗口显示程序运行参数
show args

# 调试命令窗口启动程序，如果程序有断点则会停在第一个断点处
run
# 可以简写为r

# 调试命令窗口启动程序并且向下执行一行代码，在第一条语句处停止
start

# 调试命令窗口查看源代码
list 
# 默认显示10行源代码，按回车会继续向下显示源代码
# list 可以简写为l
# list [行号] 查看当前源代码文件从指定的行号开始查看源代码
# list [源文件名称:行号] 查看指定源代码文件的指定行源代码
# list [函数名] 查看当前源代码指定函数名的源代码
# list [源文件名称:函数名] 查看指定源代码文件的指定函数的源代码

# 修改显示的源代码行数，例如显示15行源代码
set listsize 15

# 查看当前list显示的代码行数
show listsize

# 设置断点
break [行号或函数名]
# break可以简写为b
# break [行号或函数名] 设置当前源代码文件指定的行号或者是函数入口处打断点
# break [源文件名称:行号或函数名] 设置指定源代码文件指定的行号或者是函数入口处打断点
# break [行号或函数名] if a==10  当前源代码文件设置条件断点，当运行中条件满足时会停在断点处
# break [源文件名称:行号] if a==10  设置指定源代码文件指定的行号处打断条件断点，当运行中条件满足时会停在断点处

# 查看设置的断点
info b 或者是 i b
# Num列表示断点编号
# Type列表示断点类型
# Enb列表表示断点是否有效
# What列表示断点是在哪个文件的多少行设置的

# 使设置的断点无效
disable [断点编号1] [断点编号2] [更多断点...]
# disable [1-5] 设置连续的一段断点编号无效

# 设置断点有效
enable [断点编号1] [断点编号2] [更多断点...]
# enable [1-5] 设置连续的一段断点编号有效

# 删除断点
delete [断点编号1] [断点编号2] [更多断点...]
# 可以简写为d
# delete [1-5] 删除连续的一段断点编号
# delete 表示是否删除全部断点

# 打印变量的值
print [变量名]
# 可以简写为p
# print [&变量] 可以打印变量的地址

# 查看变量的类型
ptype [变量名]

# 修改变量的值，注意这可能导致出现一些问题，请不要在生产环境下使用
set var [变量名]=[变量值]

# 设置变量监视，每次执行代码都会直到显示变量的值
display [变量名]

# 查看变量监视列表
info display 或者是 i display
# Num列表示监视的变量编号
# Enb列表表示变量监视是否有效
# Expression列表示监视的变量名称

# 设置变量监视无效
undisplay [变量编号1] [变量编号2] [更多变量编号...]
# 或者是 disable display [变量编号1] [变量编号2] [更多变量编号...]
# 当然上述两种方式都可以使用 连续设置变量监视无效 例如 undisplay 1-5 1到5号变量监视无效

# 删除变量监视
delete display [变量编号1] [变量编号2] [更多变量编号...]
# 连续删除 例如 delete 1-5 删除1到5号变量监视

# 单步跟踪
next
# 可以简写为n
# 会将函数调用当成一条语句执行并不会进入函数调用内部
# 在第一次输入next后，后续如果需要继续next，直接回车就行了

# 单步跟踪
step
# 可以简写为s
# 发生函数调用时会进入函数调用内部
# 在第一次输入step后，后续如果需要继续step，直接回车就行了

# 退出当前进入的函数
finish
# 注意当前函数中后续是否有其他的断点或者是循环中是否有其他断点，如果有则无法退出需要删除断点或者是设置无效断点

# 退出循环体
until
# 可以简写为u
# 在一个循环体中循环次数过多会导致单步跟踪次数过多，这时可以使用until跳出循环体
# 注意当前循环体中是否有其他断点，如果有则无法退出需要删除断点或者是设置无效断点

# 跳到下一个断点
continue
# 可以简写为c
# 继续运行程序直到下一个断点处停止或者是程序结束



# 退出gdb调试
quit
# 询问是否退出调试

~~~

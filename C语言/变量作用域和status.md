~~~C
// https://cplusplus.com/reference/ c语言库函数查看
#include<stdio.h>  // 引入头文件stdio.h  printf函数必须引入该文件

// static修饰的变量存在内存的静态区，不会随着作用域销毁，只有当程序终止时才会销毁
// 在代码块中声明相同的static变量不会重复创建，而是使用同一内存空间的数据，当然内存空间中的数据可能会发生变化
// static修饰的全局变量和函数无法通过关键字extern在其他源文件引入了
// (本质是一种封装，防止其他文件引入后乱改全局变量的值或者是胡乱调用函数)
// 原因是static将全局变量的外部链接属性变成了内部链接属性

int a = 100;

int main() {
	int a = 10;
	printf("当全局变量与局部变量的变量名称相同时优先使用局部变量，例如a的值是：%d\n",a);

	//const修饰的常量必须被初始化并且不允许修改，但是其实本质还是变量只是不能被修改，例如数组创建时不能使用const修饰的常量作为数组长度
	const b = 888;  

	return 0; // 标准返回0代表程序正常返回，其他返回表示程序异常
}
~~~

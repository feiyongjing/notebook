~~~C
// https://cplusplus.com/reference/ c语言库函数查看
#include<stdio.h>  // 引入头文件stdio.h  printf函数必须引入该文件

// 宏定义           # define 宏名 常量 其实是进行替换
// 取消定义好的宏   # undef 宏名 取消定义好的宏后 之后的代码就都不能使用还宏名来进行替换了
// 注意声明宏的语句最后绝对不要添加分号，否则会当成宏定义的常量来进行预处理代码替换，导致出现语法错误产生奇怪的bug
// 带参数的宏定义，函数名参数（参数没有类型限制）和函数体 不过不是执行完函数体后返回而是函数体代码替换
# define π 3.1415926
# define A 3+4    
// #参数名 可以获取传递参数时的参数变量名的字符串
# define FUN(a,b){printf("带参数的宏定义函数体打印："#a"是%d,"#b"是%d\n",a,b);}

# define F(a,b) a+b
# define AD(a,b) a*b
# define F1(a,b) ((a)+(b))

int main() {
    printf("宏定义常量π：%f\n",π);
	int x = 222, y = 333;
	FUN(x, y);
	printf("宏定义的本质是替换，而且是先替换再参与运算，所以结果是2+1*1+2=%d\n", AD(F(2, 1), F(1, 2)));
	printf("宏定义的本质是替换，而且是先替换再参与运算，所以结果是((2)+(3))*((2)+(6))=%d\n", AD(F1(2, 3), F1(2, 6)));

	return 0; // 标准返回0代表程序正常返回，其他返回表示程序异常

}
~~~
